    title = malta.26.t
    desc = malta.26.d
    picture = GFX_

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            set_variable = { coastal_raid_jewelry_loot = 0 }
            randomize_variable = {
                var = coastal_raid_jewelry_loot
                distribution = uniform
                min = current_min_jewelry_coastal_raid
                max = current_max_jewelry_coastal_raid
            }
            round_variable = coastal_raid_jewelry_loot

            set_variable = { coastal_raid_diamond_loot = 0 }
            randomize_variable = {
                var = coastal_raid_diamond_loot
                distribution = uniform
                min = current_min_diamond_coastal_raid
                max = current_max_diamond_coastal_raid
            }
            round_variable = coastal_raid_diamond_loot

            add_manpower = temp_manpower_coastal_raid
            add_equipment_to_stockpile = {
                type = convoy
                amount = temp_convoy_coastal_raid
            }

            set_variable = { coastal_raid_bitcoin = 0 }
            randomize_variable = {
                var = coastal_raid_bitcoin
                distribution = uniform
                min = 0
                max = 0.6
            }
            multiply_variable = { coastal_raid_bitcoin = mission_naval_efficiency }

            clr_country_flag = MLT_ongoing_coastal_raid_mission
            set_variable = { coastal_raid_days_remaining = 0 }
        }
    }

    # Result
    option = {
        name = malta.26.a

            factor = 100

        custom_effect_tooltip = MLT_add_diamond_to_reserve_coastal_raid
        if = {
            limit = {
                check_variable = {
                    var = mission_naval_efficiency
                    value = 2
                    compare = less_than
                }
            }
            add_to_variable = { mission_naval_efficiency = 0.03 }
            custom_effect_tooltip = MLT_gain_naval_efficiency
        }
        if = {
            limit = {
                check_variable = {
                    var = coastal_raid_bitcoin
                    value = 0.5
                    compare = greater_than_or_equals
                }
            }
            add_to_variable = { bitcoin_currency = 1 }
            custom_effect_tooltip = MLT_add_bitcoin_to_reserve
            if = {
                limit = {
                    NOT = {
                        has_country_flag = MLT_bitcoin_discovered
                    }
                }
                set_country_flag = MLT_bitcoin_discovered
            }
        }
        clr_country_flag = MLT_ongoing_coastal_raid_mission_event

        if = {
            limit = {
                NOT = {
                    has_country_flag = MLT_research_slot_found
                }
            }
            set_variable = { randomized_factor_research_slot = 0 }
            randomize_temp_variable = {
                var = randomized_factor_research_slot
                distribution = uniform
                min = 1
                max = 50
            }
            round_variable = randomized_factor_research_slot

            if = {
                limit = {
                    check_variable = {
                        var = research_slot_tries
                        value = randomized_factor_research_slot
                        compare = greater_than_or_equals
                    }
                }
                set_country_flag = MLT_research_slot_found
                country_event = { id = malta.46 days = 2 }

                add_to_variable = { research_slot_tries = 1 }
            }
        }

                    value = 2
                    compare = less_than
                }
            }
            add_to_variable = { mission_naval_efficiency = 0.03 }
            custom_effect_tooltip = MLT_gain_naval_efficiency
        }
        if = {
            limit = {
                check_variable = {
                    var = coastal_raid_bitcoin
                    value = 0.5
                    compare = greater_than_or_equals
                }
            }
            add_to_variable = { bitcoin_currency = 1 }
            custom_effect_tooltip = MLT_add_bitcoin_to_reserve
            if = {
                limit = {
                    NOT = {
                        has_country_flag = MLT_bitcoin_discovered
                    }
                }
                set_country_flag = MLT_bitcoin_discovered
            }
        }

        set_country_flag = mlt_custom_gui_tower_opened
        custom_effect_tooltip = MLT_open_custom_GUI

        if = {

                    has_country_flag = MLT_research_slot_found
                }
